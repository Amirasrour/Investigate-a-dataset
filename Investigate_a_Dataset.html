def myGroPlot(df, grpVar, yVar=None, title=None, ylabel=None, rotation=None, figsize=(14,6), alpha=0.7):
    '''base function for groupby plotting of dependent variables to no_show column'''
    # plotting
    df[show].groupby([grpVar])[yVar].mean().plot.bar(edgecolor='black', figsize=figsize, rot=0, width=0.6, color='blue', alpha=alpha, label='show')
    df[noshow].groupby([grpVar])[yVar].mean().plot.bar(edgecolor='black', figsize=figsize, rot=0, width=0.6, color='red', alpha=alpha,label ='noshow')
    plt.title(f' {title}'.title(), fontsize = 14, weight = "bold")
    plt.xlabel(grpVar.title(), fontsize =8, weight = "bold")
    plt.ylabel(f'{ylabel}'.title(), fontsize = 8, weight='bold')
    plt.xticks(rotation=rotation)
    plt.grid(axis='y')
    plt.legend();
